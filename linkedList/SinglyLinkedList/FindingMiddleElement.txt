/* Node of a linked list
 class Node {
   int data;
    Node next;
    Node(int d)  { data = d;  next = null; }
}
*/

class Solution
{
    int getMiddle(Node head)
    {
         // Your code here.
         int middle=0;
         int count=0;
         Node current=head;
         while(current!=null)
         {
             current=current.next;
             count++;
         }
         if(count%2!=0)
         {
         middle=(count/2);    
         }
         if(count%2==0) 
         {
             middle=(count/2);
         }
         count=0;
         current=head;
         while(current!=null)
         {
             if(count==middle) return current.data;
             count++;
             current=current.next;
         }
         
       return 0;
    }
}
