class Node
{
    int data;
    Node next;
    Node(int key)
    {
        data = key;
        next = null;
    }
}*/
// complete the below function
class Solution
{
    public static int count(Node head, int search_for)
    {
        //code here
        Node current=head;
        int count=0;
        while(current!=null)
        {
            if(current.data==search_for) count++;
            current=current.next;
        }
        return count;
    }
}
