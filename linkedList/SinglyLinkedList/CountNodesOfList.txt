class Solution
{
    //Function to count nodes of a linked list.
    public static int getCount(Node head)
    {
        
        //Code here
        int count=0;
        Node current=head;
        while(current!=null)
        {
            current=current.next;
            count++;
        }
        return count;
    }
}
