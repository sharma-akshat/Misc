/* Structure of class Node is
class Node
{
	int data;
	Node next;
	
	Node(int d)
	{
		data = d;
		next = null;
	}
}*/

class Solution
{
    //Function to check whether the list is palindrome.
    boolean isPalindrome(Node head) 
    {
        //Your code here
        Stack<Integer> s=new Stack<Integer>();
        Node list=head;
        while(list!=null)
        {
        s.push(list.data);
        list=list.next;
        }
        list=head;
        int top=0;
        while(list!=null)
        {
            top=s.pop();
            if(top==list.data)
            {
                list=list.next;
            }
            else return false;
        }
        return true;
    }
}
