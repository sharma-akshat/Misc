/* Node is defined as

class Node
{
    int data;
    Node next;
    Node(int d) {data = d; next = null; }
}

*/


class Solution {
    //Function to check if the linked list has a loop.
    public static boolean detectLoop(Node head){
        // Add code here
        HashSet<Node> h=new HashSet<Node>();
        Node n=head;
        while(n!=null)
        {
            if(h.contains(n)) return true;
            h.add(n);
            n=n.next;
        }
        return false;
    }
}
